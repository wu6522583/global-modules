my97 datePicker使用

html

< input type='text' class='Wdate input_text_date'/>

JS

var datePicker = require("datePicker");

datePicker = new datePicker();

datePicker.getMy97DatePicker();

/*上面这三行代码是用来模块化加载my97DatePicker插件，项目中只需加载一次。后面要使用my97DatePicker插件只需 直接使用WdatePicker方法就行了。 */

WdatePicker({

dateFmt : "yyyy-MM-dd HH:mm:ss",

minDate : "#F{$dp.$D('createDateStart')}"

});

 My97DatePicker插件具体使用，参考官方网站。
